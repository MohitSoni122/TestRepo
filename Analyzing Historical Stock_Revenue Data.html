def make_graph(stock_data, revenue_data, stock):
    fig = make_subplots(rows=2, cols=1, shared_xaxes=True, subplot_titles=("Historical Share Price", "Historical Revenue"), vertical_spacing = .3)
    stock_data_specific = stock_data[stock_data.Date <= '2021-06-14']
    revenue_data_specific = revenue_data[revenue_data.Date <= '2021-04-30']
    fig.add_trace(go.Scatter(x=pd.to_datetime(stock_data_specific.Date, infer_datetime_format=True), y=stock_data_specific.Close.astype("float"), name="Share Price"), row=1, col=1)
    fig.add_trace(go.Scatter(x=pd.to_datetime(revenue_data_specific.Date, infer_datetime_format=True), y=revenue_data_specific.Revenue.astype("float"), name="Revenue"), row=2, col=1)
    fig.update_xaxes(title_text="Date", row=1, col=1)
    fig.update_xaxes(title_text="Date", row=2, col=1)
    fig.update_yaxes(title_text="Price ($US)", row=1, col=1)
    fig.update_yaxes(title_text="Revenue ($US Millions)", row=2, col=1)
    fig.update_layout(showlegend=False,
    height=900,
    title=stock,
    xaxis_rangeslider_visible=True)
    fig.show()
    from IPython.display import display, HTML
    fig_html = fig.to_html()
    display(HTML(fig_html))

# Step-by-step scraping of Tesla revenue (assumes html_data exists)

from bs4 import BeautifulSoup
import pandas as pd
import re

# Parse HTML (use "html.parser" or "html5lib" if you have it)
soup = BeautifulSoup(html_data, "html.parser")

# 1) Create an empty list (we'll convert to DataFrame later)
rows = []

# 2) Find the relevant table
table = None

# Try to find table by heading text (preferred)
for header in soup.find_all(['h1','h2','h3','h4']):
    if "Tesla Quarterly Revenue" in header.get_text(strip=True):
        table = header.find_next("table")
        break

# Fallback 1: find first table that mentions "Revenue"
if table is None:
    for t in soup.find_all("table"):
        if "Revenue" in t.get_text():
            table = t
            break

# Fallback 2: last table on the page
if table is None:
    tables = soup.find_all("table")
    if tables:
        table = tables[-1]

if table is None:
    raise ValueError("Tesla revenue table not found on the page.")

# 4) Iterate through rows in the table body (or all tr if tbody missing)
tbody = table.find("tbody")
tr_rows = tbody.find_all("tr") if tbody else table.find_all("tr")

for tr in tr_rows:
    cols = tr.find_all("td")
    # 5) Extract data from columns (need at least 2 columns: Date, Revenue)
    if len(cols) >= 2:
        date_text = cols[0].get_text(strip=True)
        revenue_text = cols[1].get_text(strip=True)

        # Clean revenue: remove $ and commas and any non-digit characters
        revenue_clean = re.sub(r"[^0-9.]", "", revenue_text)

        # 6) Append to list if valid
        if date_text and revenue_clean:
            rows.append({"Date": date_text, "Revenue": revenue_clean})

# Convert list to DataFrame and final cleaning
tesla_revenue = pd.DataFrame(rows)
tesla_revenue["Revenue"] = pd.to_numeric(tesla_revenue["Revenue"], errors="coerce")
tesla_revenue.dropna(subset=["Revenue"], inplace=True)
tesla_revenue.reset_index(drop=True, inplace=True)

# Show the first / last few rows to verify
print(tesla_revenue.head())
print(tesla_revenue.tail())
